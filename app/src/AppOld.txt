interface CardProps {
  suit: number, //heart, dim, club, spade
  num: number, //2 to Ace
  hidden: boolean,
  position?: number,
  total?: number
}

export const Card: FC<CardProps> = (props) => {
  const xPos = props.hidden ? 0 : props.num
  const yPos = props.hidden ? 0.8 : props.suit
  
  var img = new Image();
  img.src = './cards.gif';
  var height = img.height/5;
  var width  = img.width/13;

  var card = new Image();
  card.src = './cards.gif';
  card.onload = () => {
    this.context.drawImage(card, xPos*width, yPos*height, width, height, 0, 0, width, height)
    //ctx.drawImage(image, sx, sy, sWidth, sHeight, dx, dy, dWidth, dHeight)
  }
  
}
